Rename to positive vs.negative feelings

change_names <- function(naming){
naming %>%
  str_replace("A", "negative") %>%
  str_replace("B", "negative") %>%
  str_replace("C", "neutral") %>%
  str_replace("D", "positive") %>%
  str_replace("E", "positive")
}

change_names2 <- function(naming){
naming %>%
  str_replace("A", "very good") %>%
  str_replace("B", "okay") %>%
  str_replace("C", "bad")
}

change_names3 <- function(naming){
naming %>%
  str_replace("A", "") %>%
  str_replace("B", "negative") %>%
  str_replace("C", "negativet sure")
}
